local function findPlayer(name)
  for _, player in pairs(Players:GetPlayers()) do
    if string.lower(player.Name):sub(1, #name) == string.lower(name) or string.lower(player.DisplayName):sub(1, #name) == string.lower(name) then
      targetplr = player
      return player
    end

  return nil
end
